// The real purpose of this is to open the Properties dialog and then click Apply to make
// sure all values returned from there have been handled properly.

var a = 1;
var b = -1;
var c = 0.1;
var d = -0.1;
var e = "str";
var f = Math.pow(2, 40);
var g = -Math.pow(2, 40);
var h = true;
var i = false;
var j = "20";

var aa = window.GetProperty("a", a);
var bb = window.GetProperty("b", b);
var cc = window.GetProperty("c", c);
var dd = window.GetProperty("d", d);
var ee = window.GetProperty("e", e);
var ff = window.GetProperty("f", f);
var gg = window.GetProperty("g", g);
var hh = window.GetProperty("h", h);
var ii = window.GetProperty("i", i);
var jj = window.GetProperty("j", j);

console.log(a === aa);
console.log(b === bb);
console.log(c === cc);
console.log(d === dd);
console.log(e === ee);
console.log(f === ff);
console.log(g === gg);
console.log(h === hh);
console.log(i === ii);
console.log(j === jj);
