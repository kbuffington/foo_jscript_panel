window.SetProperty("2K3.TEXT.FONTS.FIXED", false);

var panel = new _.panel();
var text = new _.text('text_reader', LM, TM, 0, 0);

var files = utils.ListFiles(fb.ComponentPath + "test\\chardet\\", true).toArray();
var i = 0;

blah(i);

function blah(index) {
	text.filename = files[index];
	text.codepage = utils.Chardet(text.filename);
	text.content = utils.ReadTextFile(text.filename, text.codepage);
	text.update();
	window.Repaint();
}

function on_size() {
	panel.size();
	text.w = panel.w - (LM * 2);
	text.h = panel.h - TM;
	text.size();
}

function on_paint(gr) {
	panel.paint(gr);
	gr.GdiDrawText(text.filename + " - " + text.codepage, panel.fonts.title, panel.colours.highlight, LM, 0, panel.w - (LM * 2), TM, LEFT);
	gr.DrawLine(text.x, text.y + 1, text.x + text.w, text.y + 1, 1, panel.colours.highlight);
	text.paint(gr);
}

function on_mouse_wheel(s) {
	i += s;
	if (i < 0) i = files.length - 1;
	if (i > files.length - 1) i = 0;
	blah(i);
}

function on_mouse_move(x, y) {
	text.move(x, y);
}

function on_mouse_lbtn_up(x, y) {
	text.lbtn_up(x, y);
}

function on_key_down(k) {
	text.key_down(k);
}

function on_mouse_rbtn_up(x, y) {
	return panel.rbtn_up(x, y, text);
}
